    #!/bin/sed -f
    #DTPD#
    #convert *bold* text to *html bold*
    # also converts /italic/ text to html em
    s/\*\([^\*]*\)\*/<strong>\1<\/strong>/g;
    s/\/\([^\/]*\)\//<em>\1<\/em>/g;
