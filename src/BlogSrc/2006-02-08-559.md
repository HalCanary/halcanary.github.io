    ## track-numbers.sh
    ## (C) 2006 Hal Canary
    ## DTPD
    for NUMBER in $(seq -w 1 99); do
        if (/bin/ls *${NUMBER}* &> /dev/null); then
            id3tag -t ${NUMBER} *${NUMBER}*
        fi
    done
