#!/bin/sh
## $HOME/Palm/pilot-xfer-install
export PILOTPORT=/dev/ttyS0
export PILOTRATE=115200
if ! (/bin/ls -l $PILOTPORT | grep -q rwxrwxrwx) ; then
    echo "sudo chmod 777 $PILOTPORT"
    sudo chmod 777 $PILOTPORT
fi
exec pilot-xfer -i "$@"
