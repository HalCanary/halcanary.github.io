#!/bin/sh
## $HOME/bin/unzip-nozipbomb
## Written 2007 Hal Canary
## DTPD ##
for zipfile in "$@" ; do
    dir=$(basename "${zipfile}" .zip)
    if [ "${dir}" = "${zipfile}" ] ; then
	dir="${dir}.d"
    fi
    mkdir -p "${dir}"
    unzip -d "${dir}" "${zipfile}"
done
