TEMP=tempfile.txt
# 'random_word' grabs N rand words
# from the dictionary file.
random_word 200 > ${TEMP}
for word in $(cat ${TEMP}) ; do
    hash=$(echo ${word} | sha1sum - ) ;
    # remeber that s oe versions of
    # cnvert are broken.
    convert -background "#d0d0d0" \
        -fill black -pointsize 20 \
        label:${word} "${hash}.png";
done
rename "  -" "" *.png
rm $TEMP
