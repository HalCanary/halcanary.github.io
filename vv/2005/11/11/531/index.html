#!/bin/sh

##  2 of the GNU GPL.
for x in "$@" ; do
  if [ -d "$x" ] ; then
    if [ "$x" != "." -a "$x" != ".." ] ; then
      touch "${x}"/EMPTY
      rm-shred "${x}"/*
      rmdir "${x}"
    fi
  else
    shred -n 1 -u "$x"
  fi
done

(back)
