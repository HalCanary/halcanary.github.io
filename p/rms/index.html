#!/bin/sh

TRASH="/tmp/$USER/trash";

if [ "$#" -lt 1 ]; then
    echo "Usage: $0 FILE...";
    echo "Move FILE(s) to $TRASH.";
    exit 1;
fi

[ -d "$TRASH" ] || mkdir -p "$TRASH";

for file in "$@"; do
    if [ -f "$file" ]; then
	mv -f "$file" "$TRASH"
	echo "$file -> $TRASH"
    else
	echo "$file doesn't exist!"
    fi
done
exit 0;
